### Install

<hr/>

```
npm i element-ui vue-ele-ui -S
```

### Quick Start

<hr/>

在 main.js 中写入以下内容：

```
import Vue from 'vue';
import ElementUI from 'element-ui';
import 'element-ui/lib/theme-chalk/index.css';
import VueEleUI from 'vue-ele-ui';
import 'vue-ele-ui/lib/vue-ele-ui.css';
import App from './App.vue';
Vue.use(ElementUI);
Vue.use(VueEleUI);
```
